any?

array = [1,"hank",true]
array.any? {|x| x = "hank"}
=> True

count

array = [1,"hank",true,"sam"]
array.count {|x| x = true}
=> 1

drop

array = [1,"hank",true,4,5]
array.drop(1)
=> [1,true,4,5]

drop_while

array = [1,"hank",true,4,5]
array.drop_while {|x| x > 3}
=> [1,"hank",true]

find

array = [1,5,"hank",true,3,12,5]
array.find {|x| x % 3 == 0}
=> 3

find_all

array = [1,5,"hank",true,3,12,5]
array.find_all {|x| x % 3 == 0}
=> [3,12]

grep

array = [1,2,3,4,5]
array.grep 3..5
=> [3,4,5]

max_by

array = ["ted","xi","tina","cali"]
array.max_by(2) {|x| x.length}
=> ["tina","cali"]

partition

array = [1,2,3,4,5]
array.partition {|x| x.even?}
=> [[2,4],[1,3,5]]

reduce

array = [1,2,3,4,5]
array.reduce(:+)
=> 15

sort

array = ["bill","lisa","frank"]
array.sort
=> ["bill","frank","lisa"]

take

array = [1,2,3,4,5]
array.take(2)
=> [1,2]

to_a

(1..7).to_a
=> [1,2,3,4,5,6,7]
